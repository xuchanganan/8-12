[分割等和子集](https://leetcode-cn.com/problems/partition-equal-subset-sum/)  

传统写法  
```
class Solution:
    def canPartition(self, nums) :
        num_sum = 0 
        for num in nums :
            num_sum += num 
        if num_sum % 2 != 0:
            return False
        
        c = num_sum // 2 
        num_len = len(nums) 
        # 初始化dp数组. 
        dp = [[0 for _ in range(c+1)] for _ in range(num_len)]
        for i in range(num_len):
            dp[i][0] = 1

        for i in range(1, c+1):
            if nums[0] == i:
                dp[0][i] = 1 

        # 开始递推. 
        for i in range(1, num_len):
            # 第 i 件商品.  
            for j in range(1, c+1):
                # 容量为 j. 
                if nums[i] > j:
                    dp[i][j] = dp[i-1][j]
                else:
                    if dp[i-1][j-nums[i]] == 1:
                        dp[i][j] = 1
                    else:
                        dp[i][j] = dp[i-1][j]

        return dp[num_len-1][c]   
```  
优化一下  
```
class Solution:
  def canPartition(self, nums):
    nums_sum = sum(nums)
    if nums_sum % 2 != 0:
        return False
    c = nums_sum // 2
    dp = [[0] * (c+1) for _ in range(2)]  
    # 对第一列数据进行处理
    dp[0][0], dp[1][0] = 1, 1
    # 对存放进来的第一个数进行处理
    if nums[0] <= c:
      dp[0][nums[0]] = 1
    for i in range(1, len(nums)):
      num = nums[i]  
      for j in range(1, len(nums)):
        num = nums[i]
        for j in range(1, c+1)
          cur = i % 2 
          if cur == 0:
            pre = 1
          else:
            pre = 0
          if num > j :
            dp[cur][j] = dp[pre][j]
            continue 
          dp[cur][j] = dp[pre][j] or dp[pre][j-num]
    return dp[-1][-1]
```
