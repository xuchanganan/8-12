**问题：一次跑两步, 不会跑超么？**  
```
def EntryNodeOfLoop(self, pHead):
        # write code here
        if pHead == None:
            return None
        
        fast = slow = pHead 
        flag = False 
        while fast.next and fast.next.next :
            # 一次走两步, 不会跑超么？ 
            slow = slow.next 
            fast = fast.next.next 
            
            if fast == slow:
                flag = True
                break
        if flag == False:
            return None 
        
        slow = pHead 
        while fast != slow:
            fast = fast.next
            slow = slow.next
        return fast
```
