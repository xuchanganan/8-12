# 递归写法.   
```
def reverseList(self, head):
        if head.next == None:
            return head
        
        # 返回的好像注定是最后一个元素.  
        ahead_node = self.reverseList(head.next)
        ahead_node.next = head 
        head.next = None 
        return head
        # 最后返回的也是最后一个元素. 
```

# 非递归写法(三指针).
三个指针正好可以解决. 
```
def reverse(self, head):
    pre = None 
    cur = head 
    nxt = head 

    while cur != None:
        nxt = cur.next 
        cur.next = pre 
        pre = cur 
        cur = nxt 
    return pre
```
