# 挖坑法  
```
def partition(self, nums, left, right):
  pivot = nums[left]
  l = left 
  r = right 
  
  while l < r:
    while l < r and nums[r] >= pivot:
      r -= 1
    if l < r:
      nums[l] = nums[r]
    while l < r and nums[l] <= pivot:
      l += 1
    if l < r:
      nums[r] = nums[l]  
  
  nums[l] = pivot 
  return l 
```

# 双指针法
```
def swap(self, nums, idx_a, idx_b):
  tmp = nums[idx_b]  
  nums[idx_b] = nums[idx_a]
  nums[idx_a] = tmp 

def partition(self, nums, left, right):
  pivot = nums[left]  
  l = left
  r = right 
  
  while l < r:
    while l < r and nums[r] >= pivot:
      r -= 1
    while l < r and nums[l] <= pivot:
      l += 1
    if l == r:
      break 
    self.swap(nums, left, right)
 
  nums[left] = nums[l]
  nums[l] = pivot
  return l  
```
