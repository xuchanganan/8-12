# 逆序单链表.  
```
def reverseList(self, head):
        if head.next == None:
            return head
        
        # 返回的好像注定是最后一个元素.  
        ahead_node = self.reverseList(head.next)
        ahead_node.next = head 
        head.next = None 
        return head
        # 最后返回的也是最后一个元素. 
```
